Vector2 usage

constructor

Vector2(x: number, y: number)
add (v: Vector2) : Vector2

---

v1 = new Vector2(2, 2);
v2 = new Vector2(2, 2);
v3 = v1.add(v2) => Vector2(4, 4)
addByNumber (n: number) : Vector2

---

v1 = new Vector2(2, 2);
v2 = v1.addByNumber(2) => Vector2(4, 4);
sub (v: Vector2) : Vector2

---

v1 = new Vector2(3, 3);
v2 = new Vector2(2, 2);
v3 = v1.sub(v2) => Vector2(1, 1)
subByNumber(n: number) : Vector2

---

v1 = new Vector2(2, 2);
v2 = v1.subByNumber(1, 1) => Vector2(1, 1)
div (v: Vector2) : Vector2

---

v1 = new Vector2(4, 4);
v2 = new Vector2(2, 2);
v3 = v1.div(v2) => Vector2(2, 2)
divByNumber (n: number) : Vector2

---

v1 = new Vector2(4, 4); v2 = v1.divByNumber(2) => Vector2(2, 2);
mul (v: Vector2) : Vector2

---

v1 = new Vector2(4, 4);
v2 = new Vector2(2, 2);
v3 = v1.mul(v2) => Vector2(8, 8)
mulByNumber (n: number) : Vector2

---

v1 = new Vector2(2, 2); v2 = v1.mulByNumber(2) => Vector2(4, 4)
length () : number

---

return the length of the vector

normalize () : Vector2

---

returns normalized vector

distance (v: Vector2) : number

---

returns distance between vectors

clone () : Vector2

---

returns copy of itself

flatten () : Vector2

---

returns vector2 flattened to x or y max 1, example:

v1 = new Vector2(2, 1); v2 = v1.flatten(); => Vector2(1, 0.5)
static between (input: Vector2, min: Vector2, max: Vector2) : boolean

---

return true if input will be between min and max

static clamp (input: Vector2, min = 0, max = 0) : Vector2

---

v1 = new Vector2(2, 0.5); v2 = Vector2.clamp(v1, 0, 1) => Vector2(1, 0.5)
static equal (in1: Vector2, in2: Vector2) : boolean

---

returns true if in1 is equal in2

static randomInRange (xmin = 0, xmax = 0, ymin = 0, ymax = 0) : Vector2

---

returns randomized Vector between inputs

static direction (from: Vector2, to: Vector2) : Vector2

---

returns direction from vector to vector
